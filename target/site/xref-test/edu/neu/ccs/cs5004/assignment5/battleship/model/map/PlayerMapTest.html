14  public class PlayerMapTest {
15    private Map example;
16    private Map sameExample;
17    private Map sameStateExample;
18    private Map anotherExample;
19    private Map diffExample;
20    private Map nullExample;
21    @Before
22    public void setUp() throws Exception {
23      example = Map.createMap();
24      sameExample = example;
25      sameStateExample = Map.createMap();
26      anotherExample = Map.createMap();
27      diffExample = Map.createMap();
28      diffExample.replaceCell(Row.R1,Col.J,Cell.createGapWaterCell());
29      nullExample = null;
30    }
31  
32    @Test
33    public void getPos() {
34      Assert.assertEquals(example.getPos(Row.R2, Col.H), Cell.createOpenSeaWaterCell());
35    }
36  
37    @Test
38    public void replaceCell() {
39      example.replaceCell(Row.R2,Col.H,Cell.createGapWaterCell());
40      Assert.assertEquals(example.getPos(Row.R2,Col.H),Cell.createGapWaterCell());
41    }
42  
43    @Test
44    public void prettyPrint() {
45      example.prettyPrint(new Printer());
46    }
47  
48    @Test
49    public void testEquals() {
50      Assert.assertTrue(example.equals(example));
51      Assert.assertTrue(example.equals(sameExample));
52      Assert.assertTrue(example.equals(sameStateExample));
53      Assert.assertTrue(sameStateExample.equals(example));
54      Assert.assertEquals(example.equals(sameStateExample), sameStateExample.equals(example));
55      Assert.assertEquals(example.equals(sameStateExample) && sameStateExample.equals(anotherExample),
56          anotherExample.equals(example));
57      Assert.assertFalse(example.equals(nullExample));
58      Assert.assertNotEquals(example, diffExample);
59    }
60  
61    @Test
62    public void testHashCode() {
63      Assert.assertEquals(example.equals(sameExample), example.hashCode() == sameExample.hashCode());
64    }
