9   public class AbstractShipTest {
10  
11    private AbstractShip ship;
12    private AbstractShip sameRefShip;
13    private AbstractShip sameStateShip;
14    private AbstractShip yetAnotherShip;
15    private AbstractShip diffShip;
16    private AbstractShip nullShip;
17  
18    @Before
19    public void setUp() throws Exception {
20      ship = new Cruiser(2);
21      sameRefShip = ship;
22      sameStateShip = new Cruiser(2);
23      yetAnotherShip = new Cruiser(2);
24      diffShip = new Battleship(4);
25      nullShip = null;
26    }
27  
28    @Test
29    public void getSize() {
30      Assert.assertEquals(new Integer(3), ship.getSize());
31      Assert.assertEquals(new Integer(4), diffShip.getSize());
32    }
33  
34    @Test
35    public void getNumOfHitCells() {
36      Assert.assertEquals(new Integer(2), ship.getNumOfHitCells());
37      Assert.assertEquals(new Integer(4), diffShip.getNumOfHitCells());
38    }
39  
40    @Test
41    public void isSunk() {
42      Assert.assertTrue(diffShip.isSunk());
43      Assert.assertFalse(ship.isSunk());
44    }
45  
46    @Test
47    public void testEquals() {
48      Assert.assertEquals(ship, ship);
49      Assert.assertEquals(ship, sameRefShip);
50      Assert.assertTrue(ship.equals(sameStateShip));
51      Assert.assertTrue(sameStateShip.equals(ship));
52      Assert.assertEquals(ship.equals(sameStateShip), sameStateShip.equals(ship));
53      Assert.assertEquals(ship.equals(sameStateShip) && sameStateShip.equals(yetAnotherShip),
54              yetAnotherShip.equals(ship));
55      Assert.assertFalse(ship.equals(nullShip));
56      Assert.assertNotEquals(ship, diffShip);
57    }
58  
59    @Test
60    public void testHashCode() {
61      Assert.assertEquals(ship.equals(sameRefShip), ship.hashCode() == sameRefShip.hashCode());
62      Assert.assertNotEquals(ship.hashCode(), diffShip.hashCode());
63      Assert.assertEquals(ship.hashCode(), yetAnotherShip.hashCode());
64    }

66  }
