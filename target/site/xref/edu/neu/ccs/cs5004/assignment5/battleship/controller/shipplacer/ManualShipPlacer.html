18  class ManualShipPlacer extends AbstractShipPlacer {
19  
20    private BufferedReader bufferedReader;
21    private Observer consoleObserver;
22  
23  
24    ManualShipPlacer() {
25      bufferedReader = new BufferedReader(new InputStreamReader(System.in, StandardCharsets.UTF_8));
26      consoleObserver = Observer.createConsoleObserver();
27    }
28  
29    @Override
30    public void placeShips(Player player) {
31      Map fleetMap = player.getFleetMap();
32      Ship[] ships = player.getShips();
33  
34      Observable mapUpdate;
35      mapUpdate = Observable.createMapUpdate(fleetMap);
36      mapUpdate.subscribeViewers(consoleObserver);
37  
38      String row;
39      String col;
40      String configuration;
41  
42      int leftNumOfShips = ships.length;
43      int index = 0;
44  
45      while (leftNumOfShips > 0) {
46  
47        try {
48  
49          while (true) {
50  
51            System.out.println("You Are Placing a " + ships[index]);
52  
53            do {
54              System.out.println("Please Enter the Row Number.");
55              row = bufferedReader.readLine();
56            } while (row == null
57                    || row.length() > 2
58                    || (row.length() == 1 && (row.compareTo("1") < 0 || row.compareTo("9") > 0))
59                    || (row.length() == 2 && !row.equals("10")));
60  
61            do {
62              System.out.println("Please Enter the Column Number.");
63              col = bufferedReader.readLine();
64            }
65            while (col == null || col.length() > 1 || col.compareTo("A") < 0 || col.compareTo("J")
66                    > 0);
67  
68            do {
69              System.out.println("Please enter the Configuration.");
70              configuration = bufferedReader.readLine();
71            }
72            while ((configuration != null) && (!configuration.equals("H") && !configuration.equals(
73                    "V")));
74  
75            if (isValidLocation(Row.values()[Integer.parseInt(row) - 1], Col.valueOf(col),
76                    ShipConfiguration.valueOf(configuration), ships[index].getSize(), fleetMap)) {
77              placeSingleShip(Row.values()[Integer.parseInt(row) - 1], Col.valueOf(col),
78                      ShipConfiguration.valueOf(configuration), ships[index++], fleetMap);
79  
80              mapUpdate.updateMap(fleetMap);
81              leftNumOfShips--;
82              break;
83  
84            } else {
85              System.out.println("Invalid Input. Please try again.");
86            }
87  
88          }
89  
90        } catch (IOException e) {
91  
92          System.out.println("");
93        }
94  
95      }
96  
97  
98    }
99  }
