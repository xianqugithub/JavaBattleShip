12  public class RandomShipPlacer extends AbstractShipPlacer {
13  
14    private Random rand;
15  
16    RandomShipPlacer() {
17      rand = new Random();
18    }
19  
20    @Override
21    public void placeShips(Player player) {
22      Map fleetMap = player.getFleetMap();
23      Ship[] ships = player.getShips();
24  
25      int index = 0;
26      while (index < ships.length) {
27        Ship curShip = ships[index];
28        Row row = Row.values()[rand.nextInt(10)];
29        Col col = Col.values()[rand.nextInt(10)];
30        ShipConfiguration configuration = ShipConfiguration.values()[rand.nextInt(10) / 5];
31  
32        if (isValidLocation(row, col, configuration, curShip.getSize(), fleetMap)) {
33          placeSingleShip(row, col, configuration, curShip, fleetMap);
34          index++;
35        }
36      }
37  
38    }
