8   import java.util.Random;

10  /**
11   * Represents the random strategy.
12   */
13  public class RandomStrategy extends AbstractStrategy {
14  
15    private Row row;
16    private Col col;
17    private Random rand;
18  
19    RandomStrategy() {
20      rand = new Random();
21      row = Row.values()[rand.nextInt(10)];
22      col = Col.values()[rand.nextInt(10)];
23    }
24  
25    @Override
26    public void attackEnemy(Player self, Player enemy) {
27  
28      Map guessMap = self.getGuessMap();
29  
30      while (!isValidPosition(guessMap, row, col)) {
31        row = Row.values()[rand.nextInt(10)];
32        col = Col.values()[rand.nextInt(10)];
33      }
34      self.attackEnemy(enemy, row, col);
35    }
36  
37  }
