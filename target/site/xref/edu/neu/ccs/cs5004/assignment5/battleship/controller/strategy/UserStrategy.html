14  /**
15   * Represent the user's strategy.
16   */
17  class UserStrategy extends AbstractStrategy {
18  
19    private BufferedReader bufferedReader;
20    private String row;
21    private String col;
22  
23    UserStrategy() {
24      bufferedReader = new BufferedReader(new InputStreamReader(System.in, StandardCharsets.UTF_8));
25    }
26  
27    @Override
28    public void attackEnemy(Player self, Player enemy) {
29  
30      Map guessMap = self.getGuessMap();
31  
32      try {
33  
34        while (true) {
35  
36          do {
37            System.out.println("Please enter the row number.");
38            row = bufferedReader.readLine();
39          }
40          while (row == null
41                  || (row.length() > 2
42                  || (row.length() == 1 && (row.compareTo("1") < 0 || row.compareTo("9") > 0))
43                  || (row.length() == 2 && !row.equals("10"))));
44  
45          do {
46            System.out.println("Please enter the column number.");
47            col = bufferedReader.readLine();
48          }
49          while (col == null || col.length() > 1 || col.compareTo("A") < 0 || col.compareTo("J")
50                  > 0);
51  
52          if (isValidPosition(guessMap, Row.values()[Integer.parseInt(row) - 1], Col.valueOf(col))) {
53            break;
54          } else {
55            System.out.println("Invalid Input. Please try again.");
56          }
57  
58        }
59  
60      } catch (IOException e) {
61  
62        System.out.println("");
63      }
64  
65      self.attackEnemy(enemy, Row.values()[Integer.parseInt(row) - 1], Col.valueOf(col));
66    }
67  }
