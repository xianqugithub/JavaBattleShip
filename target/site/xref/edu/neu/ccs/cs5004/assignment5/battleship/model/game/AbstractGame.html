11  /**
12   * Represent a abstract game.
13   */
14  abstract class AbstractGame implements Game {
15    Player player1;
16    Player player2;
17  
18    @Override
19    public Player getPlayer1() {
20      return player1;
21    }
22  
23    @Override
24    public Player getPlayer2() {
25      return player2;
26    }
27  
28    @Override
29    public void initialize() {
30  
31      ShipPlacer humanShipPlacer;
32      final ShipPlacer computerShipPlacer;
33      String input;
34      BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in,
35              StandardCharsets.UTF_8));
36      System.out.println("Press M For Manual Setup of Ships.");
37      System.out.println("Press R For Random Setup of Ships.");
38  
39      try {
40        input = bufferedReader.readLine();
41  
42        while ((input != null) && (!input.equals("M") && !input.equals("R"))) {
43          System.out.println("Wrong Input.");
44          System.out.println("Press M For Manual Setup of Ships.");
45          System.out.println("Press R For Random Setup of Ships.");
46          input = bufferedReader.readLine();
47        }
48        if (input == null) {
49          input = "";
50        }
51        switch (input) {
52          case "M":
53            humanShipPlacer = ShipPlacer.createManualShipPlacer();
54            humanShipPlacer.placeShips(player1);
55            break;
56          case "R":
57            humanShipPlacer = ShipPlacer.createRandomShipPlacer();
58            humanShipPlacer.placeShips(player1);
59            break;
60          default:
61        }
62      } catch (IOException e) {
63        System.out.println("The input is not valid.");
64      }
65  
66      computerShipPlacer = ShipPlacer.createRandomShipPlacer();
67  
68      computerShipPlacer.placeShips(player2);
69    }
70  }
