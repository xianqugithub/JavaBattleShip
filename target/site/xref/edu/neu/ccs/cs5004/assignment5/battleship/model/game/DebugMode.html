10  import java.nio.charset.StandardCharsets;

12  /**
13   * Represent the debug mode of game.
14   */
15  public class DebugMode extends AbstractGame {
16    DebugMode() throws IOException {
17  
18      int numOfBattleship = 0;
19      int numOfCruiser = 0;
20      int numOfSubmarine = 0;
21      int numOfDestroyer = 0;
22  
23      String battleShipInput;
24      String cruiserInput;
25      String submarineInput;
26      String destroyerInput;
27  
28      System.out.println("This is the debug mode.");
29      BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in,
30              StandardCharsets.UTF_8));
31  
32      try {
33        do {
34          System.out.println("Please enter the number of battleship.");
35          System.out.println("Can not be less than 0 or more than 1.");
36          battleShipInput = bufferedReader.readLine();
37        }
38        while ((battleShipInput != null) && (!battleShipInput.equals("0") && (!battleShipInput
39                .equals("1"))));
40  
41        if (battleShipInput != null) {
42          numOfBattleship = Integer.parseInt(battleShipInput);
43        }
44  
45        do {
46          System.out.println("Please enter the number of cruiser.");
47          System.out.println("Can not be less than 0 or more than 2.");
48          cruiserInput = bufferedReader.readLine();
49        }
50        while (cruiserInput != null && ((cruiserInput.length() > 1 || cruiserInput.compareTo("0")
51                < 0 || cruiserInput.compareTo("2") > 0)));
52        if (cruiserInput != null) {
53          numOfCruiser = Integer.parseInt(cruiserInput);
54        }
55  
56        do {
57          System.out.println("Please enter the number of submarine.");
58          System.out.println("Can not be less than 0 or more than 3.");
59          submarineInput = bufferedReader.readLine();
60        }
61        while (submarineInput != null && ((submarineInput.length() > 1 || submarineInput.compareTo(
62                "0") < 0 || submarineInput.compareTo("3") > 0)));
63        if (submarineInput != null) {
64          numOfSubmarine = Integer.parseInt(submarineInput);
65        }
66  
67  
68        do {
69          System.out.println("Please enter the number of destroyer.");
70          System.out.println("Can not be less than 0 or more than 4.");
71          destroyerInput = bufferedReader.readLine();
72        }
73        while (destroyerInput != null && ((destroyerInput.length() > 1 || destroyerInput.compareTo(
74                "0") < 0 || destroyerInput.compareTo("4") > 0)));
75        if (destroyerInput != null) {
76          numOfDestroyer = Integer.parseInt(destroyerInput);
77        }
78  
79      } catch (IOException e) {
80        System.out.println("Exception thrown during deciding number of ships");
81      }
82  
83      Ship[] ships = new Ship[numOfBattleship + numOfCruiser + numOfSubmarine + numOfDestroyer];
84      int index = 0;
85      for (int i = 0; i < numOfBattleship; i++) {
86        ships[index] = Ship.createBattleShip();
87        index++;
88      }
89      for (int i = 0; i < numOfCruiser; i++) {
90        ships[index] = Ship.createCruiser();
91        index++;
92      }
93      for (int i = 0; i < numOfSubmarine; i++) {
94        ships[index] = Ship.createSubmarine();
95        index++;
96      }
97      for (int i = 0; i < numOfDestroyer; i++) {
98        ships[index] = Ship.createDestroyer();
99        index++;
100     }
101     this.player1 = Player.createHumanPlayer(ships);
102     this.player2 = Player.createComputerPlayer(ships);
103   }
