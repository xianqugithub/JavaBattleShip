11  public interface Player {
12  
13    /**
14     * Creator of new human player.
15     *
16     * @param ships that are given to the human player.
17     * @return a new human player.
18     */
19    static Player createHumanPlayer(Ship[] ships) {
20      return new HumanPlayer(ships);
21    }
22  
23    /**
24     * Creator of new computer player.
25     *
26     * @param ships that are given to the computer player.
27     * @return a new computer player.
28     */
29    static Player createComputerPlayer(Ship[] ships) {
30      return new ComputerPlayer(ships);
31    }
32  
33    /**
34     * Attack of enemy at specific position.
35     *
36     * @param enemy the player to be attacked.
37     * @param row the attacked row number [1-10].
38     * @param col the attacked col number [A-J].
39     */
40    void attackEnemy(Player enemy, Row row, Col col);
41  
42    /**
43     * Attacked at specific position.
44     *
45     * @param row the attacked row number [1-10].
46     * @param col the attacked col number [A-J].
47     */
48    void attacked(Row row, Col col);
49  
50    /**
51     * Attack Result after attacked at specific position.
52     *
53     * @param row the attacked row number [1-10].
54     * @param col the attacked col number [A-J].
55     * @return the attacked response.
56     */
57    AttackResult attackResponse(Row row, Col col);
58  
59    /**
60     * If this player is lost.
61     *
62     * @return a boolean variable that indicates if the current player is lost.
63     */
64    Boolean isLost();
65  
66    /**
67     * Getter of current player's fleet map.
68     *
69     * @return the player's fleet map.
70     */
71    Map getFleetMap();
72  
73    /**
74     * Getter of current player's guess map.
75     *
76     * @return the player's guess map.
77     */
78    Map getGuessMap();
79  
80    /**
81     * Getter of current player's ships.
82     *
83     * @return the ships that are originally assigned to the player.
84     */
85    Ship[] getShips();
86  
87    /**
88     * Getter of current player's number of ships.
89     *
90     * @return the number of ships.
91     */
92    int getNumOfShips();
93  
94    /**
95     * Getter if current player can attack.
96     *
97     * @return if the current player can attack its opponent.
98     */
99    boolean canAttack();
100 
101   /**
102    * Mutator of current player's attack status.
103    */
104   void resetAttack();
105 
106 
107 }
